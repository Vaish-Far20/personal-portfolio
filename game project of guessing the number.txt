#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int
main ()
{
  int number, guess, nguesses = 1;
  srand (time (0));
  number = rand () % 100 + 1;//creating random number

  do
    {
      printf ("Enter the number between 1 to 100\n");
      scanf ("%d", &guess);
      if (number > guess)
	{
	  printf ("larger number please\n");
	}
      else if (number < guess)
	{
	  printf ("smaller number please\n");
	}
      else
	{
	  printf ("You guessed the correct number in %d guesses\n", nguesses);
	}
      nguesses++;


    }
  while (guess != number);

  return 0;
}
